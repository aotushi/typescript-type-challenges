```ts twoslash
type AppendToObject<T, U extends string, V> = T extends {[index: string]: any}
  ? {[K in keyof T|U]: K extends keyof T ? T[K] : V}
  : T;


```

```ts twoslash
type AppendToObject<T, U extends keyof any, V> = {
  [K in keyof T | U]: K extends keyof T ? T[K] : V;
};
```
